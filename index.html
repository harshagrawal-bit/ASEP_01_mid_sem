# Sample code to control your Tapo plug
from PyP100 import PyP110

ip = "YOUR_PLUG_IP"
email = "YOUR_TAPO_EMAIL"
password = "YOUR_TAPO_PASSWORD"

p110 = PyP110.P110(ip, email, password)
p110.handshake()
p110.login()

# Turn on
p110.turnOn()

# Turn off
p110.turnOff()

# Get device info
print(p110.getDeviceInfo())
